Note: You need to set the width and height attribute of the images to prevent a bug in Chrome.

Code

$(document).ready(function() {
  $('#photos').photostack({
    speed : 600,
    easeIn : 'easeOutExpo',
    easeOut : 'easeInExpo',
    direction : 'horizontal',
    randomTop : 0,
    randomLeft : 50,
    rotateDeg : 2
  });
});
